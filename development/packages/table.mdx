The `@conar/table` package provides a virtualized table component optimized for rendering large database query results with smooth scrolling and dynamic column sizing.

## Core Concepts

### Virtual Scrolling

The table uses TanStack Virtual to render only visible rows and columns:

- **Rows**: Only renders rows in viewport + overscan
- **Columns**: Only renders columns in viewport + overscan
- **Performance**: Handles 100,000+ rows smoothly

### Context-Based

All table state is managed via React Context:

```typescript
interface TableContextType {
  scrollRef: RefObject<HTMLDivElement | null>
  scrollDirection: ScrollDirection
  rows: Record<string, unknown>[]
  columns: ColumnRenderer[]
  virtualRows: VirtualItem[]
  virtualColumns: VirtualItem[]
  tableHeight: number
  tableWidth: number
}
```

## Basic Usage

```typescript
import { 
  TableProvider, 
  Table, 
  TableHeader, 
  TableBody 
} from '@conar/table'

function QueryResults({ data }: { data: Record<string, unknown>[] }) {
```

**Implementation:**

From `packages/table/src/provider.tsx:12`:

```typescript
export function TableProvider({
  rows,
  columns,
  children,
  estimatedRowSize = DEFAULT_ROW_HEIGHT,
  estimatedColumnSize = DEFAULT_COLUMN_WIDTH,
  customColumnSizes,
}: TableProviderProps) {
  const scrollRef = useRef<HTMLDivElement | null>(null)
  const scrollDirection = useScrollDirection(scrollRef)

  // Vertical virtualizer for rows
  const { getVirtualItems: getVirtualRows, getTotalSize: getTableHeight } = 
    useVirtualizer({
      count: rows.length,
      getScrollElement: () => scrollRef.current,
      estimateSize: () => estimatedRowSize,
      overscan: verticalScroll ? 10 : 0,
    })

  // Horizontal virtualizer for columns
  const { getVirtualItems: getVirtualColumns, getTotalSize: getTableWidth } = 
    useVirtualizer({
      horizontal: true,
      count: columns.length,
      getScrollElement: () => scrollRef.current,
      estimateSize: index => 
        customColumnSizes?.[columns[index]!.id] ?? 
        columns[index]!.size ?? 
        estimatedColumnSize,
      overscan: horizontalScroll ? 3 : 0,
    })

  // ... context provider
}
```

## Selection Hooks

### useShiftSelectionClick

Handles shift+click range selection.

```typescript
import { useShiftSelectionClick } from '@conar/table'

function SelectableTable() {
  const [selected, setSelected] = useState<Set<number>>(new Set())
  
  const handleClick = useShiftSelectionClick({
    onSelectionChange: (indices) => {
      setSelected(new Set(indices))
    },
  })
  
  return (
    <TableBody
      onRowClick={(rowIndex, event) => handleClick(rowIndex, event)}
    />
  )
}
```

### useShiftSelectionKeyDown

Handles keyboard navigation and selection.

```typescript
import { useShiftSelectionKeyDown } from '@conar/table'

function KeyboardTable() {
  const [selection, setSelection] = useState<SelectionState>({
    anchor: null,
    selected: new Set(),
  })
  
  const handleKeyDown = useShiftSelectionKeyDown({
    totalItems: rows.length,
    selection,
    onSelectionChange: setSelection,
  })
  
  return (
    <div onKeyDown={handleKeyDown}>
      <TableBody />
    </div>
  )
}
```

## Dynamic Column Sizing

Columns can be resized dynamically:

```typescript
function ResizableTable() {
  const [columnSizes, setColumnSizes] = useState<Record<string, number>>({
    id: 80,
    name: 200,
  })
  
  const handleResize = (columnId: string, newSize: number) => {
    setColumnSizes(prev => ({ ...prev, [columnId]: newSize }))
  }
  
  return (
    <TableProvider
      rows={rows}
      columns={columns}
      customColumnSizes={columnSizes}
    >
      <Table onColumnResize={handleResize}>
        <TableHeader />
        <TableBody />
      </Table>
    </TableProvider>
  )
}
```

## Performance Tips

### Optimize Re-renders

Use selector functions with `useTableContext`:

```typescript
// Good ✓
const rowCount = useTableContext(ctx => ctx.rows.length)

// Bad ✗ - Re-renders on any context change
const { rows } = useContext(TableContext)
const rowCount = rows.length
```

### Memoize Column Definitions

```typescript
const columns = useMemo<ColumnRenderer[]>(() => [
  { id: 'id', size: 80 },
  { id: 'name', size: 200 },
], [])
```

### Estimated Sizes

Provide accurate size estimates for better scrollbar behavior:

```typescript
<TableProvider
  estimatedRowSize={40}  // Actual row height
  estimatedColumnSize={180} // Average column width
>
```

## Real-World Example

Complete example with custom renderers:

```typescript
import { 
  TableProvider, 
  Table, 
  TableHeader, 
  TableBody,
  type ColumnRenderer,
  type TableCellProps 
} from '@conar/table'

function BooleanCell({ value, style }: TableCellProps) {
  return (
    <div style={style} className="px-4 py-2">
      {value ? '✓' : '✗'}
    </div>
  )
}

function NumberCell({ value, style }: TableCellProps) {
  return (
    <div style={style} className="px-4 py-2 text-right">
      {Number(value).toLocaleString()}
    </div>
  )
}

function QueryResultsTable({ results }: { results: any[] }) {
  const columns: ColumnRenderer[] = useMemo(() => {
    const firstRow = results[0]
    if (!firstRow) return []
    
    return Object.entries(firstRow).map(([key, value]) => {
      let cell: ColumnRenderer['cell']
      
      if (typeof value === 'boolean') {
        cell = BooleanCell
      } else if (typeof value === 'number') {
        cell = NumberCell
      }
      
      return {
        id: key,
        size: 150,
        cell,
      }
    })
  }, [results])
  
  return (
    <TableProvider rows={results} columns={columns}>
      <Table className="w-full h-[600px] border">
        <TableHeader />
        <TableBody />
      </Table>
    </TableProvider>
  )
}
```
