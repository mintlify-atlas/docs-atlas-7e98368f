## Encryption

AES-256-GCM encryption for connection strings and sensitive data.

### encrypt()

```typescript
function encrypt(params: { text: string; secret: string }): string
```

**Usage:**

```typescript
import { encrypt } from '@conar/shared/utils/encryption'

const encrypted = encrypt({
  text: 'postgresql://user:pass@localhost/db',
  secret: 'user-password',
})

console.log(encrypted)
// 'a1b2c3d4e5f6.9f8e7d6c5b4a.1a2b3c4d5e6f'
```

**Implementation:**

From `packages/shared/src/utils/encryption.ts:9`:

```typescript
import { Buffer } from 'node:buffer'
import * as crypto from 'node:crypto'

const ALGORITHM = 'aes-256-gcm'
const IV_LENGTH = 12
const KEY_LENGTH = 32
const SALT_LENGTH = 16

export function encrypt({ text, secret }: { text: string, secret: string }) {
  const iv = crypto.randomBytes(IV_LENGTH)
  const salt = crypto.randomBytes(SALT_LENGTH)

  const key = crypto.pbkdf2Sync(secret, salt, 1, KEY_LENGTH, 'sha256')

  const cipher = crypto.createCipheriv(ALGORITHM, key, iv)
  const encrypted = Buffer.concat([
    cipher.update(text, 'utf8'),
    cipher.final(),
    cipher.getAuthTag(),
  ])

  return `${iv.toString('hex')}.${encrypted.toString('hex')}.${salt.toString('hex')}`
}
```

### decrypt()

```typescript
function decrypt(params: { encryptedText: string; secret: string }): string
```

**Usage:**

```typescript
import { decrypt } from '@conar/shared/utils/encryption'

try {
  const decrypted = decrypt({
    encryptedText: 'a1b2c3d4e5f6.9f8e7d6c5b4a.1a2b3c4d5e6f',
    secret: 'user-password',
  })
  console.log(decrypted) // 'postgresql://user:pass@localhost/db'
} catch (error) {
  console.error('Decryption failed:', error.message)
}
```

**Security Features:**

- AES-256-GCM authenticated encryption
- PBKDF2 key derivation (1 iteration for performance)
- Random IV per encryption (prevents ciphertext reuse)
- Random salt per encryption
- Auth tag verification (prevents tampering)

## Utilities

### SafeURL

Safe URL parsing that handles connection strings with special characters.

```typescript
import { SafeURL } from '@conar/shared/utils/safe-url'

const url = new SafeURL('postgresql://user:p@ss@localhost:5432/db')

console.log(url.username) // 'user'
console.log(url.password) // 'p@ss'
console.log(url.hostname) // 'localhost'
console.log(url.port)     // '5432'
console.log(url.pathname) // '/db'
```

```typescript
  test('encrypts and decrypts correctly', () => {
    const secret = 'test-secret'
```
